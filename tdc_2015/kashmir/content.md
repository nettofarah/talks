```ruby
class Person
  include Kashmir

  def initialize(name, age)
    @name = name
    @age = age
  end

  representations do
    rep :name
    rep :age
  end
end

Person.new('Netto Farah', 27).represent(:name, :age)

=> {:name=>"Netto Farah", :age=>"27"}
```

    rep(:title)
    rep(:chef)
  end

  include Kashmir

  representations do
    rep(:name)
  end

netto = Chef.new(
  name: 'Netto Farah'

beef_stew.represent(
  :title,
  { :chef => :name }

  :chef => {
